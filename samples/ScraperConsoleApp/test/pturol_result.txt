$(document).ready(function() {
    const bannerLinks = $('.banner-link');
    var currentIndex = 0;
    var bannerInterval;

    bannerLinks.hide();
    bannerLinks.eq(currentIndex).fadeIn(200);

    function startBannerAnimation() {
        if (bannerLinks.length <= 1) {
            return; // Przerywamy animację, jeśli ilość banerów wynosi 1 lub mniej
        }

        bannerInterval = setInterval(function() {
            bannerLinks.eq(currentIndex).fadeOut(1000);
            currentIndex = (currentIndex + 1) % bannerLinks.length;
            bannerLinks.eq(currentIndex).fadeIn(1000);
        }, 5000);
    }

    function stopBannerAnimation() {
        clearInterval(bannerInterval);
    }

    bannerLinks.on("mouseenter", stopBannerAnimation);
    bannerLinks.on("mouseleave", startBannerAnimation);

    startBannerAnimation();

    const bannerLinksMobile = $('.banner-link-mobile');
    var currentIndexMobile = 0;
    var bannerIntervalMobile;

    bannerLinksMobile.hide();
    bannerLinksMobile.eq(currentIndexMobile).fadeIn(200);

    startBannerAnimationMobile();
